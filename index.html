import datetime import json import pandas as pd import streamlit as st # --- CONFIGURATION & PAGE SETUP --- st.set_page_config( page_title="Academic OS - Multi-Profile Platform", page_icon="⬢", layout="wide", ) # Custom styling to resemble the original modern UI st.markdown( """ """, unsafe_allow_html=True, ) # --- SESSION STATE INITIALIZATION --- # Replicates your JS localStorage array if "data_store" not in st.session_state: st.session_state.data_store = [] if "streak" not in st.session_state: st.session_state.streak = 0 if "last_logged_date" not in st.session_state: st.session_state.last_logged_date = None if "is_admin" not in st.session_state: st.session_state.is_admin = False # --- STREAK LOGIC --- def process_streak_logic(date_input): today = datetime.date.today() yesterday = today - datetime.timedelta(days=1) if st.session_state.last_logged_date != date_input: if ( st.session_state.last_logged_date == yesterday and date_input == today ): st.session_state.streak += 1 elif st.session_state.last_logged_date != today: st.session_state.streak = 1 st.session_state.last_logged_date = date_input # --- NAVBAR / TOP BANNER --- st.title("⬢ ACADEMIC.OS Platform") # Top Metrics Row (Streak & Role Authorization) top_col1, top_col2 = st.columns([3, 1]) with top_col1: st.markdown( f"🔥 **Streak Counter:** `{st.session_state.streak}` Days", unsafe_allow_html=True, ) with top_col2: # Admin Panel Gatekeeper if not st.session_state.is_admin: with st.popover("🔒 Student Mode (Click to Auth)"): st.markdown("### Admin Authorization") user_input = st.text_input("Operator Username") pass_input = st.text_input("Security Password", type="password") if st.button("Decrypt & Unlock Panels"): if user_input == "username" and pass_input == "password": st.session_state.is_admin = True st.rerun() else: st.error("Authentication Failure: Invalid Parameters.") else: if st.button("🔓 Root Active (Click to Lock)"): st.session_state.is_admin = False st.rerun() st.markdown("---") # --- DATA PROCESSING & DIAGNOSTICS --- df = pd.DataFrame(st.session_state.data_store) if not df.empty: df["duration"] = pd.to_numeric(df["duration"]) gross_runtime = df["duration"].sum() completed_df = df[df["status"] == "Completed"] backlog_df = df[df["status"] == "In Progress"] completeness_index = int((len(completed_df) / len(df)) * 100) module_density = df["subject"].nunique() avg_diagnostic = int(df["duration"].mean()) backlog_minutes = backlog_df["duration"].sum() else: gross_runtime = 0 completeness_index = 0 module_density = 0 avg_diagnostic = 0 backlog_minutes = 0 # --- MAIN DASHBOARD CARDS --- card1, card2, card3, card4 = st.columns(4) with card1: st.metric(label="Gross Runtime", value=f"{gross_runtime} mins") with card2: st.metric(label="Completeness Index", value=f"{completeness_index}%") st.progress(completeness_index / 100) with card3: st.metric(label="Module Density", value=module_density) with card4: st.metric(label="Avg Diagnostic", value=f"{avg_diagnostic}m") st.caption(f"⚠️ Backlog Array: {backlog_minutes}m") # --- WORKSPACE BODY LAYOUT --- col_left, col_right = st.columns([1, 2]) # LEFT COLUMN: Inputs & System Control with col_left: st.subheader("➕ Append Lesson Entry") with st.form("lesson_form", clear_on_submit=True): in_subject = st.text_input( "Course / Module Identifier", placeholder="e.g., Computer Science IV" ) in_topic = st.text_input( "Target Lesson / Element Title", placeholder="e.g., Dijkstra's Algorithm" ) in_duration = st.number_input( "Allocation (Mins)", min_value=1, value=45 ) in_date = st.date_input("Timeline Stamp", value=datetime.date.today()) in_status = st.selectbox( "Execution State", ["In Progress", "Completed"] ) submit_btn = st.form_submit_button("Execute Logic Commit") if submit_btn and in_subject and in_topic: process_streak_logic(in_date) new_node = { "date": str(in_date), "subject": in_subject, "topic": in_topic, "duration": int(in_duration), "status": in_status, } st.session_state.data_store.append(new_node) st.success("Entry committed successfully!") st.rerun() # Administrative Section (Conditional Visibility) if st.session_state.is_admin: st.markdown("### ⚙️ Administrative Root Control") # System Wipe if st.button("🗑️ Deep Clean Storage Registers", type="primary"): st.session_state.data_store = [] st.session_state.streak = 0 st.session_state.last_logged_date = None st.success("Local database cleared.") st.rerun() # Inject Demo Framework if st.button("🪄 Inject Sandbox Demo Framework"): st.session_state.data_store = [ { "date": "2026-06-12", "subject": "Quantum Computing", "topic": "Entanglement Topologies", "duration": 120, "status": "Completed", }, { "date": "2026-06-15", "subject": "Discrete Mathematics", "topic": "Graph Isomorphisms", "duration": 90, "status": "Completed", }, { "date": "2026-06-18", "subject": "Neural Architecture", "topic": "Backpropagation Scaling", "duration": 150, "status": "In Progress", }, { "date": "2026-06-21", "subject": "Quantum Computing", "topic": "Shor's Processing Parameters", "duration": 80, "status": "Completed", }, ] st.session_state.streak = 4 st.rerun() # JSON Export Backup if not df.empty: json_string = json.dumps(st.session_state.data_store, indent=2) st.download_button( label="📥 Export Full Backup Array (JSON)", file_name="academic_os_system_export.json", mime="application/json", data=json_string, ) # RIGHT COLUMN: Progress Charts & Active Ledger Table with col_right: st.subheader("📊 Module Completion Velocity") if not df.empty: # Calculate completion rate per subject velocity_data = [] for subject, group in df.groupby("subject"): total_items = len(group) done_items = len(group[group["status"] == "Completed"]) rate = int((done_items / total_items) * 100) velocity_data.append({"Subject": subject, "Completion %": rate}) v_df = pd.DataFrame(velocity_data) st.bar_chart(data=v_df, x="Subject", y="Completion %") else: st.info("No structural metrics compiled into metrics register.") st.subheader("🗄️ Active Ledger Array") # Filters f_col1, f_col2 = st.columns(2) with f_col1: search_query = st.text_input("Scan records...", placeholder="Filter...") with f_col2: filter_state = st.selectbox("Filter Status", ["All", "In Progress", "Completed"]) # Apply Filter Transformations to DataFrame Display if not df.empty: display_df = df.copy() if search_query: display_df = display_df[ display_df["subject"].str.contains(search_query, case=False) | display_df["topic"].str.contains(search_query, case=False) ] if filter_state != "All": display_df = display_df[display_df["status"] == filter_state] if not display_df.empty: # Formatting for display st.dataframe( display_df[["date", "subject", "topic", "duration", "status"]], use_container_width=True, hide_index=True, ) else: st.warning("No structural nodes found matching parameters.") else: st.warning("No structural nodes found matching parameters.")
